PREFIX countries: <http://example.org/countries/>
PREFIX imdb: <http://example.org/movieDB/>
prefix gdp: <http://example.org/gdp/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

#change to INSERT
CONSTRUCT {?country owl:sameAs ?cGeoNames} 

WHERE {
    ?cGeoNames countries:name ?name; countries:isoCode ?isoCode; countries:iso3Code ?iso3Code; countries:fipsCode ?fipsCode.
    ?movie a imdb:Movie;
           imdb:country ?country.
    {?movie imdb:country/rdfs:label ?isoCode}
    UNION
    {?movie imdb:country/rdfs:label ?iso3Code}
    UNION
    {?movie imdb:country/rdfs:label ?fipsCode}
    UNION
    {?movie imdb:country/rdfs:label ?name}
} LIMIT 100
